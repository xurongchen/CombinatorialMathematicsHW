利用单纯形法求下列问题的解.
\begin{enumerate}
    \item 
    $\max z = 2x_1+x_2$
    \begin{align*}
        \begin{cases}
            x_1 + 4x_2 \le 32\\
            x_1 + x_2 \le 11\\
            5x_1 + x_2 \le 35
        \end{cases}\\
        x_1,x_2 \ge 0
    \end{align*}
    \item 
    $\max z = x_1+x_2$
    \begin{align*}
        \begin{cases}
            4x_1 + 5x_2 \le 10\\
            5x_1 + 2x_2 \le 10\\
            3x_1 + 8x_2 \le 12
        \end{cases}\\
        x_1,x_2 \ge 0
    \end{align*}
    \item 
    $\max z = 2x_1+2x_2$
    \begin{align*}
        \begin{cases}
            -x_1 + x_2 \le 1\\
            x_1 + x_2 \le 3\\
        \end{cases}\\
        x_1,x_2 \ge 0
    \end{align*}
    \item 
    $\max z = x_1+x_2$
    \begin{align*}
        \begin{cases}
            -x_1 + 2x_2 \le 2\\
            x_1 - x_2 \ge 2
        \end{cases}\\
        x_1,x_2 \ge 0
    \end{align*}
\end{enumerate}
