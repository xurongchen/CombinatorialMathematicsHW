\begin{solution}
    % 被3整除的个数：$\lfloor500/3\rfloor=166$；\\
    % 被5整除的个数：$\lfloor500/5\rfloor=100$；\\
    被3*5=15整除的个数：$\lfloor500/15\rfloor=33$；\\
    % 被3*7=21整除的个数：$\lfloor500/21\rfloor=23$；\\
    % 被5*7=35整除的个数：$\lfloor500/35\rfloor=14$；\\
    被3*5*7=105整除的个数：$\lfloor500/105\rfloor=4$；\\
    于是被3和5整除但不被7整除的数的个数：
    $33-4=29$.
    % $166+100-33-23-14+4=200$.
\end{solution}
